Changing CSS Properties with JavaScript

Syntax: document.getElementById("idName").style.property = "new style";

Example

<p>May the <span id="force">Force </span> be with you.<p>
<button type="button" onclick="force();">Feel the Force!</button>

function force(){
document.getElementById("force").style.color="red";
document.getElementById("force").style.fontWeight="bold";
document.getElementById("force").style.fontSize="2em"; }

May the Force be with you.

Example

<div id="square"></div>
<button type="button" onclick="toParallel();">Morph Me!</button>
<button type="button" onclick="toSquare();">Back to Square</button>

function toParallel() {
document.getElementById("square").style.transform = "skew(20deg)";
document.getElementById("square").style.transition = "all 2s"; }

function toSquare() { document.getElementById("square").style.transform = "none"; }

Example

This example shows using user input and JavaScript to change CSS.

<input type="text" id="size" name="size" placeholder="25px">
<button type="button" onclick="elephant();">Submit</button>
<p id="change">Elephant</p>

function elephant() {
var size = document.getElementById("size").value;
document.getElementById("change").style.fontSize = size; }
